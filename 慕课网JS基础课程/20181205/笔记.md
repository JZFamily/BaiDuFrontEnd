eummm 或许应该研究下哈，并将input之类的和通常的标签不一样

```html
<!DOCTYPE html>
<html>
 <head>
  <title> 事件</title>  
  <script type="text/javascript">
   function count(){
      
    //获取第一个输入框的值
    var lvalue = document.getElementById("txt1");
	//获取第二个输入框的值
	var rvalue = document.getElementById("txt2");
	//获取选择框的值
	var opt = document.getElementById("select");
 
	//获取通过下拉框来选择的值来改变加减乘除的运算法则
	var rslValue;
	var lint = parseInt(lvalue.value);
   var rint = parseInt(rvalue.value);
   alert("hehe"+lint +"&nasp" +opt.innerHTML +"&nasp"+opt.value);
	switch(opt.value)
	{
      
	    case "+":
	        rslValue = lint + rint;
	        break;
	    case "-":
	        rslValue = lint - rint;
	        break;
	    case "*":
	        rslValue = lint * rint;
	        break;
	    case "/":
	        rslValue = lint / rint;
	        break;
       default:
       alert("hehe"+lint +"&nasp" +opt.innerHTML);
	    break;
	}
    //设置结果输入框的值 
    alert("hehe"+rslValue);
    var rsl = document.getElementById("fruit");
    rsl.value = rslValue;
    
   }
  </script> 
 </head> 
 <body>
   <input type='text' id='txt1' /> 
   <select id='select'>
		<option value='+'>+</option>
		<option value="-">-</option>
		<option value="*">*</option>
		<option value="/">/</option>
   </select>
   <input type='text' id='txt2' /> 
   <input type='button' value=' = ' onclick="count()"/> 
   <!--通过 = 按钮来调用创建的函数，得到结果--> 
   <input type='text' id='fruit' />   
 </body>
</html>
```
